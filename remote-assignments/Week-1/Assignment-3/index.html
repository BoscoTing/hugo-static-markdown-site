def find_max(numbers):
  max_number = 0
  for i in numbers:
      if i > max_number:
          max_number = i
  return max_number

def find_position(numbers, 
                  target):
    success, fail, t = 0, 0, 0
    for num in numbers:
        if num == target:
            success += 1 
            break
        if num != target:
            fail += 1
        t += 1

    if success >= 1:
